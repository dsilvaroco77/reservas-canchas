from bs4 import BeautifulSoup # Rutas de archivos html_path = "/mnt/data/index (6).html" logo_path = "logo_uctenis_v03.png" # Asumimos que el archivo de logo ya está cargado en el mismo directorio # Cargar HTML original with open(html_path, "r", encoding="utf-8") as f: soup = BeautifulSoup(f, "html.parser") # Agregar favicon en favicon_link = soup.new_tag("link", rel="icon", href=logo_path, type="image/png") soup.head.append(favicon_link) # Ajustar paleta para que combine con el logo style_tag = soup.new_tag("style") style_tag.string = """ :root { --uctenis-green: #c6d300; --uctenis-dark: #003e46; } body { background-color: var(--uctenis-dark); color: white; } a, h1, h2, h3, h4, h5, h6 { color: var(--uctenis-green); } footer { background-color: var(--uctenis-dark); color: white; padding: 10px; text-align: center; } """ soup.head.append(style_tag) # Agregar logo en el encabezado (visible en PC y móvil) header_logo_div = soup.new_tag("div", style="text-align:center; margin:10px 0;") logo_img = soup.new_tag("img", src=logo_path, alt="Logo UCTenis Club", style="max-width:150px; height:auto;") header_logo_div.append(logo_img) # Insertar el logo al principio del if soup.body: soup.body.insert(0, header_logo_div) # Agregar footer con leyenda y logo footer_tag = soup.new_tag("footer") footer_logo = soup.new_tag("img", src=logo_path, alt="Logo UCTenis Club", style="max-width:50px; vertical-align:middle; margin-right:8px;") footer_tag.append(footer_logo) footer_tag.append(" UCTenis Club · Universidad Católica de Temuco · Desde 2024") soup.body.append(footer_tag) # Guardar HTML modificado output_path = "/mnt/data/index_uctenis_logo_simple.html" with open(output_path, "w", encoding="utf-8") as f: f.write(str(soup)) output_path
